<script>
(() => {
  const $ = s => document.querySelector(s);

  // ── Elementos ──
  const file = $('#file');
  const btnUpload = $('#btnUpload');
  const btnPaste = $('#btnPaste');
  const url = $('#url');
  const btnLoad = $('#btnLoad');
  const size = $('#size');
  const sizeVal = $('#sizeVal');
  const tileSize = $('#tileSize');
  const tileVal = $('#tileVal');
  const emojiSet = $('#emojiSet');
  const emojisWrap = $('#emojis');
  const btnSelAll = $('#btnSelAll');
  const btnSelNone = $('#btnSelNone');
  const customEmoji = $('#customEmoji');
  const btnAddEmoji = $('#btnAddEmoji');
  const brightness = $('#brightness');
  const brightVal = $('#brightVal');
  const contrast = $('#contrast');
  const contrastVal = $('#contrastVal');
  const saturation = $('#saturation');
  const satVal = $('#satVal');
  const filterSelect = $('#filterSelect');
  const bgSelect = $('#bgSelect');
  const bgColor = $('#bgColor');
  const autoToggle = $('#autoToggle');
  const btnConvert = $('#btnConvert');
  const out = $('#out');
  const ph = $('#ph');
  const cw = $('#cw');
  const drop = $('#drop');
  const info = $('#info');
  const btnDownload = $('#btnDownload');
  const btnDownloadJpg = $('#btnDownloadJpg');
  const btnCopy = $('#btnCopy');
  const btnClear = $('#btnClear');
  const miniInfo = $('#miniInfo');

  // ── Paletas de Emoji (cores calculadas automaticamente — mais fiel) ──
  const EMOJI_SETS = {
    classico: ['😀','❤️','⭐','🎮','📻','🔥','💧','🌙','☀️','🍀','🍕','🎈'],
    rostos:   ['😀','😁','😂','😍','😎','🥳','😢','😡','🤔','😴','😱','🥰'],
    natureza: ['🌳','🌸','☀️','🌊','🍂','⛰️','🌵','🌈','❄️','🍁','🌻','🌴'],
    comida:   ['🍎','🍌','🍇','🥑','🍕','🍫','🍉','🍓','🥕','🍩','🍊','🍋'],
    animais:  ['🐶','🐱','🦊','🐼','🐸','🐵','🦁','🐨','🐷','🐧','🐢','🦋'],
    cores:    ['🟥','🟧','🟨','🟩','🟦','🟪','⬛','⬜','🟫','🔴','🟡','🟢'],
  };

  let presetKey = 'classico';
  let presetList = [...EMOJI_SETS[presetKey]];
  let customList = []; // emojis adicionados manualmente, persistem entre trocas de conjunto
  let selected = new Set(presetList);

  function activeList() {
    const merged = [...presetList];
    customList.forEach(e => { if (!merged.includes(e)) merged.push(e); });
    return merged;
  }

  // ── Cor média real do emoji (renderiza e calcula a média de pixels) ──
  const colorCache = new Map();
  const colorProbe = document.createElement('canvas');
  colorProbe.width = 32; colorProbe.height = 32;
  const probeCtx = colorProbe.getContext('2d', { willReadFrequently: true });

  function computeEmojiColor(emoji) {
    if (colorCache.has(emoji)) return colorCache.get(emoji);
    probeCtx.clearRect(0, 0, 32, 32);
    probeCtx.textAlign = 'center';
    probeCtx.textBaseline = 'middle';
    probeCtx.font = '28px "Apple Color Emoji","Segoe UI Emoji","Noto Color Emoji",system-ui,sans-serif';
    try { probeCtx.fillText(emoji, 16, 17); } catch (e) {}
    let r=0, g=0, b=0, n=0;
    try {
      const data = probeCtx.getImageData(0, 0, 32, 32).data;
      for (let i = 0; i < data.length; i += 4) {
        const a = data[i+3];
        if (a < 20) continue;
        r += data[i] * a; g += data[i+1] * a; b += data[i+2] * a; n += a;
      }
    } catch (e) {}
    const col = n === 0 ? [140,140,140] : [Math.round(r/n), Math.round(g/n), Math.round(b/n)];
    colorCache.set(emoji, col);
    return col;
  }

  // ── Parse de emojis colados (lida com sequências compostas) ──
  function splitEmojis(str) {
    str = (str || '').trim();
    if (!str) return [];
    if (typeof Intl !== 'undefined' && Intl.Segmenter) {
      try {
        const seg = new Intl.Segmenter('en', { granularity: 'grapheme' });
        return [...seg.segment(str)].map(s => s.segment).filter(s => s.trim().length > 0);
      } catch (e) {}
    }
    return [...str].filter(s => s.trim().length > 0);
  }

  // ── Render dos pills de emoji ──
  function renderPills() {
    emojisWrap.innerHTML = '';
    activeList().forEach(e => {
      const b = document.createElement('button');
      b.type = 'button';
      b.className = 'pill';
      b.textContent = e;
      b.setAttribute('aria-pressed', selected.has(e) ? 'true' : 'false');
      b.onclick = () => {
        if (selected.has(e)) {
          if (selected.size > 1) selected.delete(e);
        } else {
          selected.add(e);
        }
        b.setAttribute('aria-pressed', selected.has(e) ? 'true' : 'false');
        scheduleAutoConvert();
      };
      emojisWrap.appendChild(b);
    });
  }

  emojiSet.onchange = () => {
    presetKey = emojiSet.value;
    presetList = [...EMOJI_SETS[presetKey]];
    selected = new Set(activeList());
    renderPills();
    scheduleAutoConvert();
  };

  btnSelAll.onclick = () => { selected = new Set(activeList()); renderPills(); scheduleAutoConvert(); };
  btnSelNone.onclick = () => {
    const list = activeList();
    selected = new Set(list.length ? [list[0]] : []);
    renderPills(); scheduleAutoConvert();
  };

  btnAddEmoji.onclick = () => {
    const items = splitEmojis(customEmoji.value);
    if (!items.length) return;
    items.forEach(e => { if (!customList.includes(e)) customList.push(e); selected.add(e); });
    customEmoji.value = '';
    renderPills();
    scheduleAutoConvert();
  };
  customEmoji.addEventListener('keydown', e => { if (e.key === 'Enter') btnAddEmoji.click(); });

  renderPills();

  // ── Imagem ──
  let img = new Image();
  img.crossOrigin = 'anonymous';
  let hasImage = false;

  function setInfo(t) { info.textContent = t; }

  function loadFile(f) {
    const u = URL.createObjectURL(f);
    setImage(u, f.name || 'Arquivo');
  }

  function setImage(src, label = 'Imagem') {
    hasImage = false;
    btnDownload.disabled = true;
    btnDownloadJpg.disabled = true;
    btnCopy.disabled = true;
    img = new Image();
    img.crossOrigin = 'anonymous';
    img.onload = () => {
      hasImage = true;
      ph.hidden = true; cw.hidden = false;
      drawOriginal();
      setInfo(`${label} • ${img.naturalWidth}×${img.naturalHeight}`);
      scheduleAutoConvert(true);
    };
    img.onerror = () => {
      alert('Não foi possível carregar a imagem. Se for URL, pode ser bloqueio de CORS. Tente enviar o arquivo.');
      setInfo('Erro ao carregar');
    };
    img.src = src;
  }

  function drawOriginal() {
    try {
      const maxW = 1100;
      const scale = Math.min(1, maxW / img.naturalWidth);
      const w = Math.max(1, Math.round(img.naturalWidth * scale));
      const h = Math.max(1, Math.round(img.naturalHeight * scale));
      out.width = w; out.height = h;
      const ctx = out.getContext('2d');
      ctx.clearRect(0, 0, w, h);
      ctx.drawImage(img, 0, 0, w, h);
    } catch (e) {
      const ctx = out.getContext('2d');
      ctx.clearRect(0, 0, out.width, out.height);
    }
  }

  // ── Auto-convert com debounce ──
  let autoTimer = null;
  function scheduleAutoConvert(immediate) {
    if (!autoToggle.checked || !hasImage) return;
    clearTimeout(autoTimer);
    autoTimer = setTimeout(convert, immediate ? 0 : 220);
  }

  function buildFilterString() {
    let f = `brightness(${brightness.value}%) contrast(${contrast.value}%) saturate(${saturation.value}%)`;
    if (filterSelect.value === 'pb') f += ' grayscale(100%)';
    if (filterSelect.value === 'sepia') f += ' sepia(100%)';
    if (filterSelect.value === 'negativo') f += ' invert(100%)';
    return f;
  }

  function resolveBg() {
    if (bgSelect.value === 'transparent') return null;
    if (bgSelect.value === 'black') return '#000000';
    if (bgSelect.value === 'white') return '#ffffff';
    return bgColor.value;
  }

  // ── CONVERSÃO PRINCIPAL: monta o mosaico de emojis ──
  function convert() {
    if (!hasImage) { return; }
    const activePalette = activeList().filter(e => selected.has(e));
    if (activePalette.length === 0) { setInfo('Selecione pelo menos um emoji.'); return; }

    const cols = parseInt(size.value, 10);
    const aspect = img.naturalWidth / img.naturalHeight;
    const rows = Math.max(10, Math.round(cols / aspect));

    // Amostra a imagem reduzida (com filtros aplicados) — é aqui que a "miniaturização" acontece
    const samp = document.createElement('canvas');
    samp.width = cols; samp.height = rows;
    const sctx = samp.getContext('2d', { willReadFrequently: true });
    sctx.imageSmoothingEnabled = true;
    sctx.imageSmoothingQuality = 'high';
    try {
      sctx.filter = buildFilterString();
      sctx.drawImage(img, 0, 0, cols, rows);
    } catch (e) {
      alert('Imagem bloqueada por CORS. Baixe e envie o arquivo local.');
      return;
    }
    const data = sctx.getImageData(0, 0, cols, rows).data;

    const palette = activePalette.map(e => ({ c: e, col: computeEmojiColor(e) }));

    // Limita o canvas final para não travar o navegador
    const maxDim = 2200;
    let tile = parseInt(tileSize.value, 10);
    if (cols * tile > maxDim) tile = Math.max(6, Math.floor(maxDim / cols));
    if (rows * tile > maxDim) tile = Math.max(6, Math.min(tile, Math.floor(maxDim / rows)));

    out.width = cols * tile;
    out.height = rows * tile;
    const ctx = out.getContext('2d');
    ctx.clearRect(0, 0, out.width, out.height);

    const bg = resolveBg();
    if (bg) { ctx.fillStyle = bg; ctx.fillRect(0, 0, out.width, out.height); }

    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.font = `${Math.floor(tile * 0.84)}px "Apple Color Emoji","Segoe UI Emoji","Noto Color Emoji",system-ui,sans-serif`;

    for (let y = 0; y < rows; y++) {
      const row = y * cols;
      for (let x = 0; x < cols; x++) {
        const i = (row + x) * 4;
        const a = data[i+3];
        if (a < 10) continue;
        const r = data[i], g = data[i+1], b = data[i+2];
        let best = palette[0].c, bestD = Infinity;
        for (const p of palette) {
          // distância perceptual ponderada (mais peso ao verde, como o olho humano)
          const dr = (r - p.col[0]) * 0.30;
          const dg = (g - p.col[1]) * 0.59;
          const db = (b - p.col[2]) * 0.11;
          const d = dr*dr + dg*dg + db*db;
          if (d < bestD) { bestD = d; best = p.c; }
        }
        ctx.fillText(best, x*tile + tile/2, y*tile + tile/2 + 0.5);
      }
    }

    ph.hidden = true; cw.hidden = false;
    setInfo(`Mosaico ${cols}×${rows} • ${out.width}×${out.height}px`);
    btnDownload.disabled = false;
    btnDownloadJpg.disabled = false;
    btnCopy.disabled = !navigator.clipboard || !window.ClipboardItem;
  }

  btnConvert.onclick = () => {
    if (!hasImage) { alert('Envie uma imagem primeiro.'); return; }
    convert();

  satVal.textContent = saturation.value + '%';
